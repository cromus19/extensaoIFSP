CREATE TABLE usuarios (
  ID int(10) NOT NULL PRIMARY KEY AUTO_INCREMENT,
  login varchar(30),
  nomeCompleto varchar(60),
  senha varchar(40),
  email varchar(60),
  CPF varchar(60),
  deficiencia tinyint(1) ,
  CID varchar(30),
  tipoVaga int(11) ,
  dataNascimento varchar(30),
  RG varchar(20),
  rua varchar(60),
  bairro varchar(30),
  cidade varchar(30),
  CEP varchar(10),
  telCasa int(11) ,
  telCel int(11) ,
  etnia int(11) ,
  relacaoIF int(11) ,
  obs varchar(60),
  ADM tinyint(1) 
);
create table curso(
    id int PRIMARY KEY AUTO_INCREMENT not null,
	nome varchar(200),
    descricao varchar(200)
);

create table edital(
    id int PRIMARY KEY AUTO_INCREMENT not null,
	professor varchar(200),
    dt_inicio date,
    dt_fim date,
    qtd_total integer,
    qtd_deficiente integer,
    qtd_negro integer,
    qtd_geral integer,
    descricao varchar(200),
    curso int,
    FOREIGN key (curso) REFERENCES curso(id) on delete CASCADE
);

create TABLE turma(
	id int PRIMARY KEY AUTO_INCREMENT not null,
    dt_fim date,
    dt_inicio date,
    horario integer,
    nome varchar(200)
);

create table inscricao(
	id int PRIMARY KEY AUTO_INCREMENT not null,
    edital int,
    usuario int,
    turma int,
    dt_inscricao datetime,
    FOREIGN key (edital) REFERENCES edital(id) on delete CASCADE,
    FOREIGN key (turma) REFERENCES turma(id) on delete CASCADE,
    FOREIGN key (usuario) REFERENCES usuarios(id) on delete CASCADE
);
